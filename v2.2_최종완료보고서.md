### 관리자 콘솔

```javascript
// 1. 현장 저장 시 안전사항 포함
function saveSite() {
    const safetyNotes = document.getElementById('safetyNotesInput').value.trim();
    await setDoc(doc(db, 'sites', siteId), {
        ...otherFields,
        safetyNotes: safetyNotes || '' // 안전사항 저장
    });
}

// 2. 엑셀 업로드 (기존 유지)
function handleExcelFile(file) {
    const reader = new FileReader();
    reader.onload = (e) => {
        const data = new Uint8Array(e.target.result);
        const workbook = XLSX.read(data, { type: 'array' });
        const rows = XLSX.utils.sheet_to_json(workbook.Sheets[workbook.SheetNames[0]]);
        validateAndSend(rows); // 검증 후 전송
    };
    reader.readAsArrayBuffer(file);
}
```

### 운전자 앱

```javascript
// 1. 사이트 데이터 로드 (안전사항 포함)
async function loadSiteDataForDispatch(dispatch) {
    const startQuery = query(
        collection(db, 'sites'),
        where('address', '==', dispatch.startLocation),
        limit(1)
    );
    const startSnap = await getDocs(startQuery);
    if (!startSnap.empty) {
        startSiteData = startSnap.docs[0].data();
        // 안전사항 배지 표시/숨김
        if (startSiteData.safetyNotes) {
            document.getElementById('startSafetyBadge').classList.remove('hidden');
        }
    }
}

// 2. GPS 기반 자동 팝업
async function checkGeofence(lat, lng) {
    const distStart = getDistance(lat, lng, curDispatch.startLat, curDispatch.startLng);
    
    // 상차지 100m 이내 진입
    if (distStart <= 0.1 && startSiteData?.safetyNotes) {
        const safetyKey = 'start_' + curDispatch.id;
        if (!shownSafetyNotes.has(safetyKey)) {
            shownSafetyNotes.add(safetyKey); // 중복 방지
            autoShowSafetyNote('start'); // 자동 팝업
        }
    }
}

// 3. 자동 팝업 (진동 + 비프음)
function autoShowSafetyNote(type) {
    navigator.vibrate([200, 100, 200]); // 진동 3회
    const beep = new Audio('data:audio/wav;base64,...'); // 비프음
    beep.play().catch(() => {});
    showSafetyNote(type); // 모달 표시
}

// 4. 수동 팝업
function showSafetyNote(type) {
    const siteData = type === 'start' ? startSiteData : endSiteData;
    document.getElementById('safetyTitle').innerText = 
        (type === 'start' ? '상차지' : '하차지') + ' 안전사항';
    document.getElementById('safetyLocation').innerText = siteData.address;
    document.getElementById('safetyText').innerText = siteData.safetyNotes;
    document.getElementById('safetyModal').classList.add('show');
}
```


```
운전자 앱: startLocation으로 sites 조회 → safetyNotes 로드
```
